        begin:
            if aux < (n + 1):
                resultFac = resultFac*aux
                aux = aux + 1
                gosub factorial
            endif
    endp

    procedure exponencial:
        begin:
            while aux < n:
                resultExp = resultExp * X
                aux = aux + 1
            endw
    endp

        print("seno")
